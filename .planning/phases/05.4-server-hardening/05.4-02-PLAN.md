    req: Request,
    next: Next,
) -> Result<Response, StatusCode> {
    let config = &state.auth_config;
    match config.mode {
        AuthMode::None => {
            let mut req = req;
            req.extensions_mut().insert(UserContext::anonymous());
            Ok(next.run(req).await)
        }
        AuthMode::ApiKey => {

                    req.extensions_mut().insert(UserContext::new(user_id, permissions, role));
                    Ok(next.run(req).await)
                }
                _ => Err(StatusCode::UNAUTHORIZED),
            }

pub const CURRENT_VERSION: u32 = 13;

pub fn migrate(conn: &Connection) -> rusqlite::Result<()> {
    let version: u32 = conn.pragma_query_value(None, "user_version", |row| row.get(0))?;
    if version < CURRENT_VERSION {
        // Race window: another process can also read user_version=0,
        // then both try to ALTER → duplicate column panic.
        info!(from = version, to = CURRENT_VERSION, "Running database migration");
        let migrations = get_migrations();
        for migration in migrations {
            if migration.version > version {
                migration.execute(conn)?;
                info!("Applied migration v{}", migration.version);
            }
        }
        conn.pragma_update(None, "user_version", CURRENT_VERSION)?;
    }
    Ok(())
}

       pub fn migrate(conn: &Connection) -> rusqlite::Result<()> {
           // BEGIN EXCLUSIVE acquires SQLite writer lock immediately (file-level).
           // Multiple processes serialize through this lock; only one runs migrations.
           // T-02 mitigation per ADR-V2-028 + NEW-A2 (forensics 2026-05-16).
           conn.execute_batch("BEGIN EXCLUSIVE")?;

           // Double-check: re-read user_version inside the txn.
           // If a prior winner already migrated, exit clean (idempotent).
           let version: u32 = conn.pragma_query_value(None, "user_version", |row| row.get(0))?;
           if version >= CURRENT_VERSION {
               conn.execute_batch("COMMIT")?;
               return Ok(());
           }

           info!(from = version, to = CURRENT_VERSION, "Running database migration (exclusive)");
           let migrations = get_migrations();
           for migration in migrations {
               if migration.version > version {
                   migration.execute(conn)?;
                   info!("Applied migration v{}", migration.version);
               }
           }
           conn.pragma_update(None, "user_version", CURRENT_VERSION)?;
           conn.execute_batch("COMMIT")?;
           info!("Migration to v{CURRENT_VERSION} complete");
