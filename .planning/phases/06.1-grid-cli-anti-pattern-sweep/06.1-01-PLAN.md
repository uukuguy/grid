    AuthFailed(String),
    SessionNotFound(String),
    NetworkError(String),
    RateLimited(String),
    StorageError(String),
    SyncError(String),
    Other(String),
}

impl GridError {
    pub fn session_not_found(id: impl Into<String>) -> Self {
        GridError::SessionNotFound(id.into())
    }
    // ... other constructors
}

impl std::error::Error for GridError {}
// `?` propagation from `Result<T, GridError>` into `Result<T, anyhow::Error>` works
// because anyhow has a blanket From<E: Error + Send + Sync + 'static> impl.

pub enum ExitCode {
    Ok = 0, General = 1, InvalidArg = 2, AuthFailed = 3,
    SessionNotFound = 4, NetworkError = 5, RateLimited = 6,
    StorageError = 7, SyncError = 73,
}

impl From<GridError> for ExitCode {
    fn from(err: GridError) -> Self {
        match err {
            GridError::SessionNotFound(_) => ExitCode::SessionNotFound,
            // ... etc
        }
    }
}
